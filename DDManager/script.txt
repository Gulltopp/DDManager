CREATE TABLE CHARACTERS
(
    id    INT PRIMARY KEY AUTO_INCREMENT,
    name	VARCHAR(30),
	playerName	VARCHAR(30),
	race	VARCHAR(30),
	characterClass VARCHAR(30),
	level INT,
	experience INT,
	strength INT,
	constitution INT,
	dexterity INT,
	intelligence INT,
	wisdom INT,
	charisma INT,
	speed INT,
	maxHp INT,
	bloodiedValue INT,
	surgesPerDay INT,
	surgeValue INT
)
